Create Kustomization
~~~~~~~~~~~~~~~~~~~~

flux create kustomization webapp-latest \
    --target-namespace=default \
    --source=webapp-latest \
    --path="./kustomize" \
    --prune=true \
    --interval=1m \
    --retry-interval=2m \
    --health-check-timeout=3m
